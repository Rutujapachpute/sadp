Q.1 Write a Java Program to implement Strategy Pattern for Duck Behavior. Create 
instance variable that holds current state of Duck from there, we just need to handle 
all Flying Behaviors and Quack Behavior [15 M]

interface FlyBehavior {
	void fly();
}

// QuackBehavior interface
interface QuackBehavior {
	void quack();
}

// Concrete Fly Behaviors
class FlyWithWings implements FlyBehavior {
	public void fly() {
		System.out.println("Flying with wings!");
	}
}

class NoFly implements FlyBehavior {
	public void fly() {
		System.out.println("I can't fly.");
	}
}

// Concrete Quack Behaviors
class QuackSound implements QuackBehavior {
	public void quack() {
		System.out.println("Quack!");
	}
}

class Silent implements QuackBehavior {
	public void quack() {
		System.out.println("<< Silence >>");
	}
}

// Duck class using Strategy pattern
class Duck {
	FlyBehavior flyBehavior;
	QuackBehavior quackBehavior;

	public Duck(FlyBehavior fly, QuackBehavior quack) {
		this.flyBehavior = fly;
		this.quackBehavior = quack;
	}

	public void performFly() {
		flyBehavior.fly();
	}

	public void performQuack() {
		quackBehavior.quack();
	}

	// You can change behavior at runtime
	public void setFlyBehavior(FlyBehavior fly) {
		flyBehavior = fly;
	}

	public void setQuackBehavior(QuackBehavior quack) {
		quackBehavior = quack;
	}
}

// Main class to test
public class Main {
	public static void main(String[] args) {
		Duck duck1 = new Duck(new FlyWithWings(), new QuackSound());
		System.out.println("Duck 1:");
		duck1.performFly();
		duck1.performQuack();

		System.out.println("\nDuck 2:");
		Duck duck2 = new Duck(new NoFly(), new Silent());
		duck2.performFly();
		duck2.performQuack();

		// Changing duck2's behavior at runtime
		System.out.println("\nChanging Duck 2's flying behavior...");
		duck2.setFlyBehavior(new FlyWithWings());
		duck2.performFly();
	}
}

/*OUTPUT
Duck 1:
Flying with wings!
Quack!

Duck 2:
I can't fly.
<< Silence >>

Changing Duck 2's flying behavior...
Flying with wings!
*/
